# Header

## User performance
```sql

-- Average handle time by yser
Select USEREXTRV, PCNAMEEXTRV, COUNT(1) as ItemCount, 
	AVG(DATEDIFF(SS, DATESTARTV, DATEEXTRV)) as AVGTime,
	SUM(DATEDIFF(SS, DATESTARTV, DATEEXTRV)) / 60 /60 as TotalTimeH,
	MIN(DATEDIFF(SS, DATESTARTV, DATEEXTRV)) as MINTime,
	MAX(DATEDIFF(SS, DATESTARTV, DATEEXTRV)) as MAXTime,
	SUM(CASE WHEN DATEDIFF(SS, DATESTARTV, DATEEXTRV) <= 30 Then 1 else 0 END) as Sec30,
	1.00 * SUM(CASE WHEN DATEDIFF(SS, DATESTARTV, DATEEXTRV) <= 30 Then 1 else 0 END)/ COUNT(1) as Per30,
	SUM(CASE WHEN DATEDIFF(SS, DATESTARTV, DATEEXTRV) <= 60 and DATEDIFF(SS, DATESTARTV, DATEEXTRV) > 30 Then 1 else 0 END) as Sec30_60,
	1.00 * SUM(CASE WHEN DATEDIFF(SS, DATESTARTV, DATEEXTRV) <= 60 and DATEDIFF(SS, DATESTARTV, DATEEXTRV) > 30 Then 1 else 0 END)/ COUNT(1) as Per30_60,
	SUM(CASE WHEN DATEDIFF(SS, DATESTARTV, DATEEXTRV) <= 90 and DATEDIFF(SS, DATESTARTV, DATEEXTRV) > 60 Then 1 else 0 END) as Sec60_90,
	1.00 * SUM(CASE WHEN DATEDIFF(SS, DATESTARTV, DATEEXTRV) <= 90 and DATEDIFF(SS, DATESTARTV, DATEEXTRV) > 60 Then 1 else 0 END)/ COUNT(1) as Per60_90,
	SUM(CASE WHEN DATEDIFF(SS, DATESTARTV, DATEEXTRV) <= 120 and DATEDIFF(SS, DATESTARTV, DATEEXTRV) > 90 Then 1 else 0 END) as Sec90_120,
	1.00 * SUM(CASE WHEN DATEDIFF(SS, DATESTARTV, DATEEXTRV) <= 120 and DATEDIFF(SS, DATESTARTV, DATEEXTRV) > 90 Then 1 else 0 END)/ COUNT(1) as Per90_120,
	SUM(CASE WHEN DATEDIFF(SS, DATESTARTV, DATEEXTRV) > 120 Then 1 else 0 END) as SecMoreThen120,
	1.00 * SUM(CASE WHEN DATEDIFF(SS, DATESTARTV, DATEEXTRV) > 120 Then 1 else 0 END)/ COUNT(1) as PerMoreThen120
from BRWdocument 
Where CURRENT_STATUS = 800
and DATEEXPORTRTS > '2017-01-01' and CLASSNAMEV <> 'VOID' and USEREXTRV is not null
--and DATEDIFF(SS, DATESTARTV, DATEEXTRV) < 1200 -- Drop anything over 20 minutes
Group BY USEREXTRV, PCNAMEEXTRV
Order By COUNT(1) DESC

```
